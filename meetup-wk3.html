1 Processing basics

Processing is a Java-based programming environment for making visualizations. A basic script has two main structures, a void setup() and a void draw(). In the former you define initialization setting, and in the latter you run time-dependent processes. The void draw() looping frequency is define using the frameRate(). The default rate is 60fps.

2 SC3 and Processing via OSC

Processing OSC communication is based on the oscP5 library.

Run this script in Processing. 

import oscP5.*;
import netP5.*;
OscP5 oscP5;

float numFlt; // global variable

void setup() {
  size(400, 300);  //P2D or P3D
  frameRate(24);
  background(0);
  smooth();

  //--network
  OscProperties properties= new OscProperties();
  //properties.setRemoteAddress("127.0.0.1", 57120);  //osc send port (to sc)
  properties.setListeningPort(47120);               //osc receive port (from sc)
  //properties.setSRSP(OscProperties.ON);  //unused
  //properties.setDatagramSize(min(BUFFERSIZE*4, 8192));
  properties.setDatagramSize(5136);  //5136 is the minimum
  oscP5= new OscP5(this, properties);
}

void oscEvent(OscMessage msg) {
  if (msg.checkAddrPattern("/SC3_to_P5")) {
    numFlt = msg.get(0).floatValue();
  }
}

void draw() {
  background(numFlt, numFlt, numFlt);
  println("POST: ", numFlt);
  // draw rect
  stroke(256-numFlt/2, 256-numFlt*abs(sin(numFlt)), 256-numFlt/4);
  strokeWeight(4);
  fill(256-numFlt/2, 256-numFlt, 256-numFlt*abs(sin(numFlt)));
  translate(width/2, height/2);
  rotate(numFlt%64);
  rect(numFlt%64, numFlt%64, numFlt*abs(sin(numFlt))%128, numFlt*abs(sin(numFlt))%128, 6);

}

s.boot;
s.meter;

(
SynthDef(\blip, { | freq = 440, amp = 0.85, att = 0.01, rel = 0.06, ffreq = 1000 |
    var sig, env, lfo;
    sig = SinOsc.ar(freq, 0, amp);
    env = EnvGen.ar(Env.perc(att, rel), doneAction:2);
    lfo = SinOsc.kr(rel * ffreq);

    Out.ar(0, Pan2.ar(RHPF.ar(sig*env, ffreq), SinOsc.kr(211*lfo)))
}).add;
)

Synth(\blip);

(
f = fork {
    loop {
        256 do: { |i|
            n.sendMsg("/SC3_to_P5", i.asFloat); // send OSC message to P5
            Synth(\blip, [\freq, 440+i, \ffreq, 1000+i*2]);
            ((i+1).reciprocal*2).wait;
        }
    }
};
)

f.stop;
